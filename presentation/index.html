Javascript

Variables & Data Types


var currentPlayer = 'X'; // String
var position = 1; // Number
var winner = null; // Null

Arrays


					var plays = ['X', 'O', 'X', 'O', 'X', 'O', 'X', 'O', 'O'];

Array Indexes


var firstPosition = plays[0];

Conditionals


if (true) {
  // do this
} else {
  // do this other thing
}

Complex Conditionals


if (currentPlayer === 'X') {
  // Hello player X
} else if (currentPlayer === 'O') {
  // Oh, hi player O
} else {
	// I don't know who you are!
}

if (currentPlayer === 'X' && winner === 'X') {
  // Hello X, you are the winner, congrats!
}
